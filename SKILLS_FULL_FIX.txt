// ONLY the SKILLS section updated version

// 1. add this state near other useState
const [skillsInput, setSkillsInput] = useState(content.skills.join(", "));

// 2. replace Skills input with this:

<input
  className="w-full rounded-xl border border-slate-300 px-4 py-2.5"
  value={skillsInput}
  onChange={(e) => {
    const value = e.target.value;
    setSkillsInput(value);

    setContent({
      ...content,
      skills: value
        .split(/[,.，、\n]/)
        .map((item) => item.trim())
        .filter(Boolean),
    });
  }}
  placeholder="Excel, Customer Service, Inventory Control"
/>
